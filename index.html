// Botones de gama del cuadro de navegación document.querySelectorAll('#gama-navegacion .btn-primary').forEach(btn=>{ btn.addEventListener('click',()=>{ if(btn.id === 'btn-inicio'){ location.reload(); // Regresa a la página principal return; } const gama = btn.getAttribute('data-gama'); mostrarProductos(productosPorGama[gama]); }); }); // Buscador en tiempo real const buscador = document.getElementById('buscador'); buscador.addEventListener('input',()=>{ const termino = buscador.value.toLowerCase(); let filtrados = []; Object.values(productosPorGama).flat().forEach(p=>{ if(p.nombre.toLowerCase().includes(termino)) filtrados.push(p); }); mostrarProductos(filtrados.length ? filtrados : [{nombre:"No se encontró producto", precio:"", img:"", specs:""}]); }); })();
